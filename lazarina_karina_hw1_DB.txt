# 6. +Вивести клієнтів банку, які обслуговуються київськими відділеннями.
SELECT
    client.FirstName,
    client.LastName,
    client.City,
    d.DepartmentCity
FROM client
    JOIN department d on client.Department_idDepartment = d.idDepartment
where d.DepartmentCity = 'kyiv';

# 7. +Вивести імена клієнтів та їхні номера телефону, погрупувавши їх за іменами.

# 8. +Вивести дані про клієнтів, які мають кредит більше ніж на 5000 тисяч гривень.
SELECT C.*,
       a.Sum
FROM client C
    JOIN application a on C.idClient = a.Client_idClient
WHERE a.Sum >5000;

# 10. Знайти кредити, які мають найбільшу суму для кожного клієнта окремо.
SELECT c.FirstName,
       c.Education,
       MAX(a.Sum)
FROM client C
    JOIN application a on C.idClient = a.Client_idClient
GROUP BY C.idClient;

# 11. Визначити кількість заявок на крдеит для кожного клієнта.

# 12. Визначити найбільший та найменший кредити.

# 13. Порахувати кількість кредитів для клієнтів,які мають вищу освіту.
SELECT
    c.FirstName,
    c.Education,
    c.Age,
    count(a.Client_idClient)
FROM client c
    join application a on c.idClient = a.Client_idClient
where c.Education = 'high'
group by c.FirstName;

# 14. Вивести дані про клієнта, в якого середня сума кредитів найвища.
SELECT
    c.FirstName,
    c.Education,
    c.Age,
    avg(a.Sum) as sum
FROM client c
         join application a on c.idClient = a.Client_idClient
group by FirstName order by sum desc limit 1;


# 15. Вивести відділення, яке видало в кредити найбільше грошей
select d.DepartmentCity,
       d.CountOfWorkers,
    sum(a.Sum) as sum
from department d
    join client c on d.idDepartment = c.Department_idDepartment
    join application a on c.idClient = a.Client_idClient
group by d.idDepartment order by sum desc limit 1;

# Знайти львівські відділення, які видали кредитів на загальну суму більше ніж 5000
select d.DepartmentCity,
       d.idDepartment,
       sum(a.Sum) as suma
from department d
    join client c on d.idDepartment = c.Department_idDepartment
    join application a on c.idClient = a.Client_idClient
where d.DepartmentCity = 'lviv'
group by idDepartment having suma>5000;

# Знайти клієнтів, які повністю погасили кредити на суму більше ніж 5000
select c.FirstName,
       c.LastName,
       a.Sum
from client c
    join application a on c.idClient = a.Client_idClient
where a.Sum>5000 and a.CreditState = 'returned';

# /*Знайти кредити, сума яких більша за середнє значення усіх кредитів*/

       a.CreditState
from application a
group by idApplication
